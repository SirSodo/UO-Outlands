while hits > 0
    for 0 to 'signtypes'
        @clearjournal
        if @findtype signtypes[] 'any' 'ground' 1 16
            clickobject 'found'
	    	headmsg 'click'
            pause 500
            if @injournal 'fairly' 'system'
                headmsg "FAIRLY WORN" 25
            endif
            if @injournal 'greatly' 'system'
                headmsg "GREATLY WORN" 25
            endif
            if @injournal 'collap' 'system'
                headmsg "IDOC!!! IDOC!!!" 25
            endif
            @ignoreobject 'found'
        endif
    endfor
    pause 200
endwhile
